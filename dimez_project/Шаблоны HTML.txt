/* Блочные элементы */
<div>          // Блочный контейнер

<p>            // Параграф (абзац) текста с отступом
<h1>,...,<h6>  // Текстовые заголовки разного уровня
<pre>          // Предварительное форматирование текста
<blockquote>   // Текстовый блок с отступами (во все стороны)

<hr>           // Горизонтальная линия по абзацу

/* Строчные элементы */
<span>    // Строчный контейнер

<a>          // Создание ссылок
<input>      // Элемент взаимодействия с позователем и опр. форму
  type=''    // Определяет тип элемента
  value=''   // Определяет значение элемента

<img>     // Отображение изображения (GIF, JPEG, PNG)

<br>      // Параграф (абзац) текста без отступа
<em>      // Отображает текст курсивным начертанием
<i>       // Отображает текст курсивным начертанием
<strong>  // Отображает текст жирным начертанием
<b>       // Отображает текст жирным начертанием
<code>    // Отображает текст моноширинным начертанием
<tt>      // Отображает текст моноширинным начертанием

<big>     // Увеличивает размер шрифта на единицу
<small>   // Уменьшает размер шрифта на единицу
<sub>     // Отображает шрифт в виде нижнего индекса
<sup>     // Отображает шрифт в виде верхнего индекса

/* Универсальные элементы */
<del>     // Помечают текст как перечеркнутый
<ins>     // Помечают текст как подчеркнутым

<audio>   // Настройка аудиозаписи
<video>   // Настройка видеоролика
<source>  // Вставляет звуковой или видеофайл

<select>  // Элемент интерфейса в виде раскрывающегося списка

/* Атрибут события */
onclick=''     // Щелчок мыши
onkeypress=''  // Нажатие клавиши
onsubmit=''    // Отправление формы
onmouseover='' // Наведение курсора мыши на элемент
onmouseout=''  // Выведение курсора мыши за пределы границ элемента
onload=''      // Полная загрузка страницы или картинки
onchange=''    // Изменение содержимого элемента, например содержимого текстового поля формы
